/* Cursor */
const cursor = document.getElementById("cursor");

/* Thunder */
let thunderOn = false;
function toggleThunder() {
    thunderOn = !thunderOn;
    const audio = document.getElementById("thunder");
    if (thunderOn) audio.play();
    else audio.pause();
}

/* Lightning */
const bg = document.getElementById("bg");
const bctx = bg.getContext("2d");

bg.width = window.innerWidth;
bg.height = window.innerHeight;

function lightning() {
    bctx.clearRect(0,0,bg.width,bg.height);

    let x = Math.random() * bg.width;
    let y = 0;

    bctx.beginPath();
    bctx.moveTo(x,y);

    for (let i=0;i<12;i++){
        x += (Math.random()-0.5)*90;
        y += bg.height/12;
        bctx.lineTo(x,y);
    }

    bctx.strokeStyle="#f5d06f";
    bctx.shadowBlur=25;
    bctx.shadowColor="#f5d06f";
    bctx.stroke();
}
setInterval(lightning, 1400);

/* Rain */
const rain = document.getElementById("rain");
const rctx = rain.getContext("2d");
rain.width = window.innerWidth;
rain.height = window.innerHeight;

let drops = Array(120).fill().map(()=>({
    x: Math.random()*rain.width,
    y: Math.random()*rain.height,
    l: Math.random()*15
}));

function drawRain(){
    rctx.clearRect(0,0,rain.width,rain.height);
    rctx.strokeStyle="rgba(255,255,255,0.2)";
    for(let d of drops){
        rctx.beginPath();
        rctx.moveTo(d.x,d.y);
        rctx.lineTo(d.x,d.y+d.l);
        rctx.stroke();

        d.y += 6;
        if(d.y > rain.height){
            d.y = 0;
            d.x = Math.random()*rain.width;
        }
    }
}
setInterval(drawRain, 30);

/* Particles */
const p = document.getElementById("particles");
const pctx = p.getContext("2d");
p.width = window.innerWidth;
p.height = window.innerHeight;

let particles = Array(60).fill().map(()=>({
    x: Math.random()*p.width,
    y: Math.random()*p.height,
    r: Math.random()*2
}));

function drawParticles(){
    pctx.clearRect(0,0,p.width,p.height);
    pctx.fillStyle="#f5d06f";

    for(let pt of particles){
        pctx.beginPath();
        pctx.arc(pt.x,pt.y,pt.r,0,Math.PI*2);
        pctx.fill();

        pt.y -= 0.3;
        if(pt.y < 0) pt.y = p.height;
    }
}
setInterval(drawParticles, 30);
